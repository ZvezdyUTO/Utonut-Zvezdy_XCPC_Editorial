# 博弈

所有者: Zvezdy

诈骗题。有这么一个结论，几个人轮流抽签，抽签顺序不会改变抽中某个物品的概率。这题的获胜要求是抽中的字符串字典序较大，那么假如前面抽中的字母都一样，那么第一个抽到大字母的人就胜利。因为两个人抽字母的概率一样，只要结果互换那么输赢也会互换，那么它们获胜的概率一样。

但我们此时需要考虑平局的情况：平局自然是抽的东西一样，如果字母总数为奇数，那么alice必赢，如果为偶数就是真平局。我们现在就需要计算平局的概率，平局的概率用全排列公式计算。

// 两参数版本的power函数
template<class T>
constexpr T power(T a, i64 b) {
    T res {1};
    while (b) {
        if (b % 2) res *= a;
        a *= a;
        b /= 2;
    }
    return res;
}

// 三参数版本的power函数
template<class T>
constexpr T power(T a, i64 b, i64 p) {
    T res {1};
    for (; b; b /= 2, a = mul(a, a, p)) {
        if (b % 2) {
            res = mul(res, a, p);
        }
    }
    return res;
}

constexpr i64 mul(i64 a, i64 b, i64 p) {
    i64 res = a * b - i64(1.L * a * b / p) * p;
    res %= p;
    if (res < 0) {
        res += p;
    }
    return res;
}

template<i64 P>
struct MInt {
    i64 x;
    constexpr MInt() : x {0} {}
    constexpr MInt(i64 x) : x {norm(x % getMod())} {}
    
    static i64 Mod;
    constexpr static i64 getMod() {
        if (P > 0) {
            return P;
        } else {
            return Mod;
        }
    }
    constexpr static void setMod(i64 Mod_) {
        Mod = Mod_;
    }
    constexpr i64 norm(i64 x) const {
        if (x < 0) {
            x += getMod();
        }
        if (x >= getMod()) {
            x -= getMod();
        }
        return x;
    }
    constexpr i64 val() const {
        return x;
    }
    constexpr MInt operator-() const {
        MInt res;
        res.x = norm(getMod() - x);
        return res;
    }
    constexpr MInt inv() const {
        return power(*this, getMod() - 2);
    }
    constexpr MInt &operator*=(MInt rhs) & {
        if (getMod() < (1ULL << 31)) {
            x = x * rhs.x % int(getMod());
        } else {
            x = mul(x, rhs.x, getMod());
        }
        return *this;
    }
    constexpr MInt &operator+=(MInt rhs) & {
        x = norm(x + rhs.x);
        return *this;
    }
    constexpr MInt &operator-=(MInt rhs) & {
        x = norm(x - rhs.x);
        return *this;
    }
    constexpr MInt &operator/=(MInt rhs) & {
        return *this *= rhs.inv();
    }
    friend constexpr MInt operator*(MInt lhs, MInt rhs) {
        MInt res = lhs;
        res *= rhs;
        return res;
    }
    friend constexpr MInt operator+(MInt lhs, MInt rhs) {
        MInt res = lhs;
        res += rhs;
        return res;
    }
    friend constexpr MInt operator-(MInt lhs, MInt rhs) {
        MInt res = lhs;
        res -= rhs;
        return res;
    }
    friend constexpr MInt operator/(MInt lhs, MInt rhs) {
        MInt res = lhs;
        res /= rhs;
        return res;
    }
    friend constexpr std::istream &operator>>(std::istream &is, MInt &a) {
        i64 v;
        is >> v;
        a = MInt(v);
        return is;
    }
    friend constexpr std::ostream &operator<<(std::ostream &os, const MInt &a) {
        return os << a.val();
    }
    friend constexpr bool operator==(MInt lhs, MInt rhs) {
        return lhs.val() == rhs.val();
    }
    friend constexpr bool operator!=(MInt lhs, MInt rhs) {
        return lhs.val() != rhs.val();
    }
    friend constexpr bool operator<(MInt lhs, MInt rhs) {
        return lhs.val() < rhs.val();
    }
};

template<>
i64 MInt<0>::Mod = 998244353; // 默认模数
using Z = MInt<0>; // 使用运行时设置的模数

vector<Z> fact, invFact;

void precomputeFactorials(int maxN) {
    fact.resize(maxN + 1);
    invFact.resize(maxN + 1);
    
    fact[0] = 1;
    for (int i = 1; i <= maxN; ++i) {
        fact[i] = fact[i - 1] * i;
    }

    invFact[maxN] = fact[maxN].inv();
    for (int i = maxN - 1; i >= 0; --i) {
        invFact[i] = invFact[i + 1] * (i + 1);
    }
}

void solve() {
    int n; cin >> n;
    int num = 0;
    int sum = 0;
    vector<int> a(n);
    for (int i = 0; i < n; ++i) {
        char now; cin >> now;
        cin >> a[i];
        num += a[i] % 2;
        sum += a[i];
    }

    if (num > 1) {
        Z ans = 1;
        cout << ans / 2 << endl;
        return;
    }
    
    sort(a.begin(), a.end());

    if (num == 1) {
        Z den = fact[sum];
        for (auto i : a) {
            den *= invFact[i];
        }

        Z mol = fact[sum / 2];
        for (auto i : a) {
            mol *= invFact[i / 2];
        }
        
        Z equ = mol / den;
        cout << (1 - equ) / 2 + equ << endl;
        return;
    }

    Z den = fact[sum];
    for (auto i : a) {
        den *= invFact[i];
    }

    Z mol = fact[sum / 2];
    for (auto i : a) {
        mol *= invFact[i / 2];
    }
    
    Z equ = mol / den;
    cout << (1 - equ) / 2 << endl;
}
