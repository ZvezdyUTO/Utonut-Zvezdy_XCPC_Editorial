因为期望值的线性性质，所以我们在对随机变量做处理以后，可化为∑(n,i=1)E(Ii)*gi。对于某块被j个矩形覆盖的范围，我们这要在那j个矩形中随便选一个就可以拿到那个范围的贡献，而对于场上任意一块像这样的面积，我们取到它们的概率都是相同的，所以我们可以用他们的总面积来套概率计算。

那最终我们就用每一块被选中的概率*这块的总面积来算出答案，但考虑到正着直接算不好算，我们可以反着算。用（1-这一块不被选中的概率）来代表这一块被选中的概率。当我们此时会拿k个矩形时，就要把包含1~k个矩形的面积的期望值挨个计算。
至于面积？打一个离散化二位前缀和求就好。

/* ★ _____                           _         ★*/

// #pragma GCC optimize(2)
// #pragma GCC optimize(3,"Ofast","inline")
#include <bits/stdc++.h>
using namespace std;
#define int long long
#define debug(x) cout<<#x<<" = "<<x<<endl
#define endl '\n'
#define fi first
#define se second
#define PII pair<int,int>
const int MODE = 998244353;
const int INF = 1e18;

void solve() {
    int n; cin >> n;

    vector<vector<int>> C(n + 1, vector<int>(n + 1, 0));
    for (int i = 0; i <= n; ++i) {
        C[i][0] = 1;
        for (int j = 1; j <= i; ++j) {
            C[i][j] = (C[i - 1][j - 1] + C[i - 1][j]) % MODE;
        }
    }

    vector<vector<int>> f(n + 1, vector<int>(4, 0));
    vector<int> x(2 * n + 1, 0); 
    vector<int> y(2 * n + 1, 0); 
    for (int i = 1; i <= n; ++i) {
        for (int j = 0; j <= 3; ++j) {
            cin >> f[i][j];
        }
        x[++x[0]] = f[i][0];
        x[++x[0]] = f[i][2];

        y[++y[0]] = f[i][1];
        y[++y[0]] = f[i][3];
    }
    sort(x.begin() + 1, x.begin() + 1 + x[0]);
    sort(y.begin() + 1, y.begin() + 1 + y[0]);
    x.erase(unique(x.begin() + 1, x.begin() + 1 + x[0]), x.end());
    y.erase(unique(y.begin() + 1, y.begin() + 1 + y[0]), y.end());
    
    // 压缩坐标
    unordered_map<int, int> A, B;
    for (int i = 1; i <= 2*n; ++i) {
        A[x[i]] = i;
        B[y[i]] = i;
    }
    vector<vector<int>> w(2 * n + 1, vector<int>(2 * n + 1, 0));
    for (int i = 1; i <= n; ++i) {
        f[i][0] = A[f[i][0]];
        f[i][1] = B[f[i][1]];
        f[i][2] = A[f[i][2]];
        f[i][3] = B[f[i][3]];

        // w的下标是压缩过的
        ++w[f[i][0]][f[i][1]];
        --w[f[i][2]][f[i][1]];
        ++w[f[i][2]][f[i][3]];
        --w[f[i][0]][f[i][3]];
    }

    for (int i = 1; i <= 2 * n; ++i) {
        for (int j = 1; j <= 2 * n; ++j) {
            w[i][j] += w[i - 1][j];
        }
    }

    vector<int> g(2 * n + 1, 0);
    for (int i = 1; i < 2*n; ++i) {
        for (int j = 1; j < 2*n; ++j) {
            w[i][j] += w[i][j - 1];

            // 以每个方块左上角为准
            g[w[i][j]] = (g[w[i][j]] + 
                (x[i + 1] - x[i]) * (y[j + 1] - y[j])
                        % MODE + MODE) % MODE;
        }
    }

    int sum = 0;
    for (int i = 1; i <= n; ++i) {
        sum = (sum + g[i]) % MODE;
    }

    vector<int> ans(n + 1, sum);
    for (int i = 1; i <= n; ++i) {
        for (int j = 1; j <= n; ++j) {
            ans[i] = (ans[i] - C[n - j][i] * g[j] % MODE + MODE) % MODE;
        }
    }

    auto NI = [&](int num) {
        int ans = 1;
        int v = MODE - 2;
        while (v > 0) {
            if (v & 1ll) {
                ans = ans * num % MODE;
            }
            num = num * num % MODE;
            v = v>>1ll;
        }
        return ans;
    };

    for (int i = 1; i <= n; ++i) {
        cout << (ans[i] * NI(C[n][i])%MODE)% MODE << endl;
    }
    
} 
signed main(){
    ios::sync_with_stdio(false);
    cin.tie(0); cout.tie(0);
    long Zvezdy = 1;
    while (Zvezdy--) {
        solve();
    }
    return 0;
}
