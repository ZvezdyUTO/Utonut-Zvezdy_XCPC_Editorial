# 信息传递

所有者: Zvezdy
标签: 图论, 深度优先搜索

其实没这么多说法，就是一个判最小环。首先来观察输入数据可以发现，这道题是n个点n条边，所以说一定存在至少一个环，为什么说至少一个环呢？因为这题也可以存在多个环，但是由于边数限制，所有的环一定都是纯纯一个简单环。

有向图判环考虑拓补排序，用vis数组标记已经搜索过或者拍出过的点，先把所有不在环上的点全部排除，接着对每个点做dfs，搜索最小的环有多长，搜一个点标记一个点以免重搜，最后输出答案即可。

```cpp
/* ★ _____                           _         ★*/

// #pragma GCC optimize(2)
// #pragma GCC optimize(3,"Ofast","inline")
#include<bits/stdc++.h>
using namespace std;
#define int long long
#define endl '\n'
#define fi first
#define se second
#define dot pair<int,int>
int n,ans;
vector<int>gra[200001];
int rd[200001];
bool vis[200001];
void dfs(int now,int endd,int s){
    vis[now]=true;
    for(auto i:gra[now]){
        if(i==endd && s){ans=min(ans,s);return;}
        if(!vis[i]) dfs(i,endd,s+1);
    }
}
void solve(){
    int n; cin>>n; ans=n;
    for(int u=1;u<=n;++u){
        int v; cin>>v;
        gra[u].push_back(v);
        ++rd[v];
    }
    queue<int>topsort;
    for(int i=1;i<=n;++i) if(!rd[i]){
        topsort.push(i);
        vis[i]=true;
    }
    while(topsort.size()){
        auto& now=topsort.front();
        topsort.pop();
        vis[now]=true;
        for(auto i:gra[now]){
            --rd[i];
            if(!vis[i] && !rd[i])
                topsort.push(i);
        }
    }
    for(int i=1;i<=n;++i)
        if(!vis[i]) dfs(i,i,1);
    cout<<ans<<endl;
}
signed main(){
    ios::sync_with_stdio(false);
    cin.tie(0); cout.tie(0);
    int t=1;
    // cin >> t;
    while(t--){
        solve();
    }
    return 0;
}
```
